1. 求字符串S中的所有字串中有多少个的子序列不包含T
解法：双指针与二分(n*m*logn)
思路：先算出总所有可能答案，查找每个左指针与它的最小合法右指针，最后减去非法区间个数


```cpp
    S = " "+S, T = " "+T;
    n = S.size()-1, m = T.size()-1; 
    for(ll i = 1; i <= n; i++){   //先存储每个不同字符的下标以便后面二分
        ll t = S[i]-'a';
        key[t].push_back(i);
    }
    ll ans = n+n*(n-1)/2, lt = 0;
    for(ll i = 1; i <= n; i++){
        if(S[i] == T[1]){  //判断该S[i]是否为T的初始点，是则处理
            ll cot = 1, rt = i;    //rt为右指针，以lt为区间左端点的，rt为最小区间右端点，区间内子序列存在T
            for(ll j = 2; j <= m; j++){
                ll tt = T[j]-'a';
                auto p = upper_bound(key[tt].begin(), key[tt].end(), rt);  //二分查找最近且比前节点大的合法下标
                if(p == key[tt].end()){ //如果某个T的下标没有合法的则说明已该字符S以i为开头的所有子串没有符合的
                    break;
                }
                rt = *p;
                cot++;
            }
            if(cot != m) break;       //检查是否完成所有T下标的检测
            ans -= (i-lt)*(n-rt+1);   //经典左区间乘右区间
            lt = i;       //更新左指针，这样乘不会重复
        }
    }
    cout << ans << endl;
```
