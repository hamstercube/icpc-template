template<class T1, class T2> bool cmin(T1 &x, const T2 &y) { if (y < x) { x = y; return 1; } return 0; }
template<class T1, class T2> bool cmax(T1 &x, const T2 &y) { if (x < y) { x = y; return 1; } return 0; }

using namespace std;
using ll = long long;
using ld = long double;
using ull = unsigned long long;
using PII = pair<int, int>;
using PLL = pair<ll, ll>;
using vei = vector<int>;
using vevei = vector<vector<int>>;


const double PI = acos(-1);

void init(){
    ;
}

void solve(){
    int n,m,k;
    cin >> n >> m >> k;
    int s,t;
    cin >> s >> t;
    vector<PII> a[n + 10];
    for(int i = 1; i <= m; i ++){
        int u,v,w;
        cin >> u >> v >> w;
        a[u].pb({v,w});
        a[v].pb({u,w});
    }
    vevei vis(n + 10, vei(k + 5));
    vevei d(n + 10, vei(k + 5, 1e18));
    auto dj = [&](int s, int t) -> int {
        priority_queue<array<int,3>, vector<array<int,3>> ,greater<array<int,3>>> q;
        d[s][0] = 0;
        q.push({0,s,0});
        while(!q.empty()){
            auto [dist,idx,cen] = q.top();
            q.pop();
            if(vis[idx][cen]) continue;
            vis[idx][cen] = 1 ;
            for(auto it : a[idx]){
                if(d[it.fi][cen] > it.se + dist){
                    d[it.fi][cen] = it.se + dist;
                    q.push({it.se + dist, it.fi, cen});
                }
                if(cen < k && d[it.fi][cen + 1] > dist){
                    d[it.fi][cen + 1] = dist;
                    q.push({dist, it.fi, cen + 1});
                }
            }
        }
        int ans = 1e18;
        for(int i = 0; i <= k; i ++){
            cmin(ans,d[t][i]);
        }
        return ans;
    };
    cout << dj(s,t) << endl;
    return;
}

signed main(){
    ios::sync_with_stdio(false);
    cin.tie(0);

    init();
    
    int _ = 1;
    // cin >> _;
    while (_--) solve();
    return 0;
}
