线段树(动态开点).md

```cpp
#include <bits/stdc++.h>
#define int unsigned long long
using namespace std;

int n, m;
const int N = 1e5 + 10;

// 动态开点数组：开大空间防止节点不够
int a[120 * N];   // 线段树区间和
int lc[120 * N];  // 左儿子编号
int rc[120 * N];  // 右儿子编号
int lz[120 * N];  // 懒标记 区间加
int idx, rt;      // idx:节点计数器  rt:根节点

// 计算  l~r 等差数列和  本题初始权值为 a[i]=i
int qwq(int l, int r) {
	return (l + r) * (r - l + 1) / 2;
}

// 向上更新：用左右子树信息更新当前节点
void pushup(int x) {
	a[x] = a[lc[x]] + a[rc[x]];
}

// 懒标记下传
void pushdown(int x, int l, int r, int mid) {
	// 有懒标记才下传
	if (lz[x]) {
		// 动态开点：没有左儿子就新建
		if (!lc[x]) lc[x] = ++idx;
		// 动态开点：没有右儿子就新建
		if (!rc[x]) rc[x] = ++idx;
		
		// 更新左子树和 + 累加懒标记
		a[lc[x]] += (mid - l + 1) * lz[x];
		lz[lc[x]] += lz[x];
		// 更新右子树和 + 累加懒标记
		a[rc[x]] += (r - mid) * lz[x];
		lz[rc[x]] += lz[x];
		
		// 清空当前节点懒标记
		lz[x] = 0;
	}
}

// 区间修改：[l,r] 加上 k
// x:当前节点  s,t:当前节点管辖区间
void updata(int &x, int l, int r, int s, int t, int k) {
	// 当前节点不存在，动态新建
	if (!x) x = ++idx;
	// 当前区间完全被覆盖，直接打标记修改
	if (l <= s && t <= r) {
		a[x] += (t - s + 1) * k;
		lz[x] += k;
		return;
	}
	int mid = s + ((t - s) >> 1);
	// 下传标记，再递归修改
	pushdown(x, s, t, mid);
	if (l <= mid) updata(lc[x], l, r, s, mid, k);
	if (r > mid)  updata(rc[x], l, r, mid + 1, t, k);
	// 回溯更新当前节点
	pushup(x);
}

// 区间查询：查询 [l,r] 总和
int query(int x, int s, int t, int l, int r) {
	// 当前区间完全在查询范围内，直接返回
	if (l <= s && t <= r) {
		return a[x];
	}
	int mid = s + ((t - s) >> 1);
	int ans = 0;
	// 先下传标记
	pushdown(x, s, t, mid);
	// 左右区间递归查询
	if (l <= mid) ans += query(lc[x], s, mid, l, r);
	if (r > mid)  ans += query(rc[x], mid + 1, t, l, r);
	return ans;
}

signed main() {
	cin >> n >> m;
	while (m--) {
		int op;
		cin >> op;
		if (op == 1) {
			// 操作1：区间 [l,r] 加上 k
			int l, r, k;
			cin >> l >> r >> k;
			updata(rt, l, r, 1, n, k);
		} else {
			// 操作2：查询 [l,r] 总和
			// 线段树存的是增量 + 原本等差数列和 = 真实答案
			int l, r;
			cin >> l >> r;
			cout << query(rt, 1, n, l, r) + qwq(l, r) << endl;
		}
	}
	return 0;
}
```
