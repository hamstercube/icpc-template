template<class T>
struct BIT {
    vector<T> a;
    int n;

    BIT() {}

    BIT(int n) {
        init(n);
    }

    void init(int n) {
        this->n = n;
        a.assign(n + 1, 0);
    }

    void add(int x, T k) {
        for (; x <= n; x += x & -x) {
            a[x] += k;
        }
    }

    void add(int x, int y, T k) {
        add(x, k);
        add(y + 1, -k);
    }

    T ask(int x) {
        T ans = 0;
        for (; x > 0; x -= x & -x) {
            ans += a[x];
        }
        return ans;
    }

    T ask(int x, int y) {
        if(y < x) return 0; 
        return ask(y) - ask(x - 1);
    }

    int kth(T k) {
        int pos = 0;
        for (int i = __lg(n); i >= 0; i--) {
            int nxt = pos + (1 << i);
            if (nxt <= n && a[nxt] < k) {
                k -= a[nxt];
                pos = nxt;
            }
        }
        return pos + 1;
    } // 第k小
};

void solve(){
    int n1, n2;
    cin >> n1 >> n2;
    vei a(n1 + n2 + 10);
    BIT<int> tr(n1 + n2 + 10);
    vector<array<ll, 2>> v;
    DF(i,n1,1){
        cin >> a[i];
        tr.add(i,1);
        v.push_back({a[i], i});
    }
    a[n1 + 1] = 0;
    F(i,n1 + 2, n1 + 2 + n2){
        cin >> a[i];
        tr.add(i,1);
        v.push_back({a[i], i});
    }
    sort(v.begin(), v.end(), greater<array<ll, 2>>());
    int p = n1+1;
    int ans = 0;
    for(auto [x, y] : v){
        int l = min(p,y) + 1;
        int r = max(p,y) - 1;
        ans += tr.ask(l,r);
//         cout << l << " " << r << " " << p << endl;
//         debug(ans);
        p = y;
        tr.add(y, -1);
    }
    cout << ans << endl;
    //F(i,1,n1 + n2 + 1) cout << a[i] << " ";
    return;
}
