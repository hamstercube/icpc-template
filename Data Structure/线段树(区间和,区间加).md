线段树(区间和,区间加)

```cpp
#include <bits/stdc++.h>
#define int long long
using namespace std;

const int N = 1e5 + 10;

// 原数组、线段树数组、懒标记数组
int a[N];
int tr[4 * N];
int lz[4 * N];

// 建树：s-t为当前区间，p为当前节点编号
void build(int s, int t, int p) {
    if (s == t) {
        tr[p] = a[s];
        return;
    }
    int m = s + ((t - s) >> 1);
    build(s, m, p << 1);
    build(m + 1, t, p << 1 | 1);
    tr[p] = tr[p << 1] + tr[p << 1 | 1];
}

// 区间更新：[l,r] += c，当前区间[s,t]，节点p
void updata(int l, int r, int c, int s, int t, int p) {
    // 当前区间完全在目标区间内，直接更新
    if (l <= s && t <= r) {
        tr[p] += (t - s + 1) * c;
        lz[p] += c;
        return;
    }
    int m = s + ((t - s) >> 1);
    // 懒标记下传
    if (lz[p] && s != t) {
        tr[p << 1] += lz[p] * (m - s + 1);
        tr[p << 1 | 1] += lz[p] * (t - m);
        lz[p << 1] += lz[p];
        lz[p << 1 | 1] += lz[p];
        lz[p] = 0;
    }
    if (l <= m) updata(l, r, c, s, m, p << 1);
    if (r > m)  updata(l, r, c, m + 1, t, p << 1 | 1);
    // 更新当前节点值
    tr[p] = tr[p << 1] + tr[p << 1 | 1];
}

// 区间查询：查询[l,r]的和，当前区间[s,t]，节点p
int getsum(int l, int r, int s, int t, int p) {
    if (l <= s && t <= r) return tr[p];
    int m = s + ((t - s) >> 1);
    // 懒标记下传
    if (lz[p]) {
        tr[p << 1] += lz[p] * (m - s + 1);
        tr[p << 1 | 1] += lz[p] * (t - m);
        lz[p << 1] += lz[p];
        lz[p << 1 | 1] += lz[p];
        lz[p] = 0;
    }
    int sum = 0;
    if (l <= m) sum = getsum(l, r, s, m, p << 1);
    if (r > m)  sum += getsum(l, r, m + 1, t, p << 1 | 1);
    return sum;
}

signed main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) cin >> a[i];
    build(1, n, 1);

    while (m--) {
        int op;
        cin >> op;
        if (op == 1) {
            // 操作1：区间 [l,r] 加上 c
            int l, r, c;
            cin >> l >> r >> c;
            updata(l, r, c, 1, n, 1);
        } else {
            // 操作2：查询区间 [l,r] 的和
            int l, r;
            cin >> l >> r;
            cout << getsum(l, r, 1, n, 1) << '\n';
        }
    }
    return 0;
}
```
