线段树(区间加乘和)

```cpp
#include<bits/stdc++.h>
#define int long long
using namespace std;

// 快读快写，考场必加
#define IOS ios::sync_with_stdio(false);cin.tie(0);cout.tie(0)

const int N=1e5+5;

int n,m,mod;       // n：数组长度 m：操作数 mod：模数
int a[N];          // 原数组
int ans[N*4];      // 线段树维护的区间和
int tag1[N*4];     // 乘法懒标记，初始为 1
int tag2[N*4];     // 加法懒标记，初始为 0

// 左儿子 = x*2
int ls(int x){return x<<1;}
// 右儿子 = x*2+1
int rs(int x){return x<<1|1;}

// 向上更新：用左右儿子更新当前节点的和
void push_up(int x){
    ans[x] = ans[ls(x)] + ans[rs(x)];
}

// 建树
void build(int x,int l,int r){
    tag1[x] = 1;   // 乘法懒标记初始为 1
    tag2[x] = 0;   // 加法懒标记初始为 0
    if(l == r){    // 叶子节点
        ans[x] = a[l];
        return;
    }
    int mid = l + r >> 1;
    build(ls(x), l, mid);    // 建左子树
    build(rs(x), mid+1, r);  // 建右子树
    push_up(x);              // 更新当前节点
}

// 核心：对当前区间执行  ans = ans * k1 + k2
// 同时更新两个懒标记（最关键的函数）
inline void f(int x,int l,int r,int k1,int k2){
    // 当前区间和 = 和 * k1 + 长度 * k2
    ans[x] = (k1 * ans[x] + (r-l+1) * k2 % mod) % mod;
    
    // 乘法标记更新：原来的乘法 * 新乘法
    (tag1[x] *= k1) %= mod;
    
    // 加法标记更新：原来的加法 * 新乘法 + 新加法
    tag2[x] = (k2 + k1 * tag2[x]) % mod;
}

// 懒标记下传（把当前节点的乘、加标记传给左右儿子）
inline void push_down(int x,int l,int r){
    int mid = l + r >> 1;
    // 下传给左儿子
    f(ls(x), l, mid, tag1[x], tag2[x]);
    // 下传给右儿子
    f(rs(x), mid+1, r, tag1[x], tag2[x]);
    // 清空当前节点标记
    tag1[x] = 1;
    tag2[x] = 0;
}

// 操作1：区间 [l,r] 全部 × k
inline void update1(int x,int nowl,int nowr,int l,int r,int k){
    // 完全覆盖，直接打标记
    if(l <= nowl && nowr <= r){
        (ans[x] *= k) %= mod;
        (tag1[x] *= k) %= mod;
        (tag2[x] *= k) %= mod;
        return;
    }
    // 先下传标记
    push_down(x, nowl, nowr);
    int mid = nowl + nowr >> 1;
    if(l <= mid) update1(ls(x), nowl, mid, l, r, k);
    if(mid < r)  update1(rs(x), mid+1, nowr, l, r, k);
    push_up(x);
}

// 操作2：区间 [l,r] 全部 + k
inline void update2(int x,int nowl,int nowr,int l,int r,int k){
    // 完全覆盖，直接打标记
    if(l <= nowl && nowr <= r){
        (ans[x] += k * (nowr-nowl+1) % mod) %= mod;
        (tag2[x] += k) %= mod;
        return;
    }
    // 先下传标记
    push_down(x, nowl, nowr);
    int mid = nowl + nowr >> 1;
    if(l <= mid) update2(ls(x), nowl, mid, l, r, k);
    if(mid < r)  update2(rs(x), mid+1, nowr, l, r, k);
    push_up(x);
}

// 查询区间 [l,r] 的和
inline int query(int x,int nowl,int nowr,int l,int r){
    if(l <= nowl && nowr <= r) return ans[x];
    int res = 0;
    // 查询前必须下传
    push_down(x, nowl, nowr);
    int mid = nowl + nowr >> 1;
    if(l <= mid) (res += query(ls(x), nowl, mid, l, r)) %= mod;
    if(mid < r)  (res += query(rs(x), mid+1, nowr, l, r)) %= mod;
    return res % mod;
}

signed main(){
    IOS;
    cin >> n >> m >> mod;
    for(int i=1;i<=n;i++) cin >> a[i], a[i] %= mod;
    build(1,1,n);
    
    for(int i=1;i<=m;i++){
        int op; cin >> op;
        if(op == 1){
            // 区间 × k
            int l,r,k; cin >> l >> r >> k;
            update1(1,1,n,l,r,k);
        }
        else if(op == 2){
            // 区间 + k
            int l,r,k; cin >> l >> r >> k;
            update2(1,1,n,l,r,k);
        }
        else{
            // 查询区间和
            int l,r; cin >> l >> r;
            cout << query(1,1,n,l,r) << endl;
        }
    }
    return 0;
}
```
