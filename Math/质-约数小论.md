质因数分解 
```cpp
void divide(int x) 
{ 
    for (int i = 2; i <= x / i; i ++ ) 
        if (x % i == 0) 
        { 
            int s = 0; 
            while (x % i == 0) x /= i, s ++ ; 
            cout << i << ' ' << s << endl; 
        } 
    if (x > 1) cout << x << ' ' << 1 << endl; 
    cout << endl; 
}
```

约数之和：
```cpp
vector<pair<int,int>> primes;
``` 
这里省略divide()代码，primes用于存储各质因子及其个数 
```cpp
long long res=1; 
for (auto p : primes){ 
     ll a = p.first, b = p.second; 
     ll t = 1; 
     while (b --)  t = (t * a + 1) % mod; 
     res = res * t % mod; 
}
```
